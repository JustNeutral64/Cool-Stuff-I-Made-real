    left(x)

def r(x):
    right(x)

def f(x):
    forward(x)

def b(x):
    backward(x)
    
def sp(x, y):
    setposition(x, y)

def c(x):
    circle(x)

def ca(x, y, z):
    circle(x, y, z)

def pd():
    pendown()

def pu():
    penup()

def sx(x):
    setx(x)

def sy(x):
    sety(x)

def sh(x):
    setheading(-x + 90)

def ps(x):
    pensize(x)

def bf():
    begin_fill()

def ef():
    end_fill()

def pc(x):
    color(x)

def bgc(x):
    bgcolor(x)

def p(x):
    print(x)

def du(x):
    setheading(90)
    forward(x)

def dl(x):
    setheading(180)
    forward(x)

def dd(x):
    setheading(270)
    forward(x)

def dr(x):
    setheading(0)
    forward(x)

    sh(x)
    forward(y)

def w(x):
    write(x)

#Sets speed to 0 for obvious reasons
speed(0)

#Defines new variables which will be used later
progress = 0
drawingnumber = -10
drawing_line = str("y")
remainder = 123
    
#Sets background color
bgc("#0d2f47")
#Gets Graph Zoom and Center Location
graphzoom = float(input("What zoom would you like your graph to be? (1 means it goes out 1 unit in every direction, 20 means it goes 20 units. etc.) "))
graphx = float(input("What would you like the x position of the center of your graph to be? "))
graphy = float(input("What would you like the y position of the center of your graph to be? "))
graphzoomdistance = 200/graphzoom

#Draws the axis
print("Drawing Graph...")
ps(4)
du(200)
b(400)
f(200)
dr(200)
b(400)
f(200)
    
#draws lines up
ps(1)
sh(0)
for i in range(int(graphzoom)):
    f(graphzoomdistance)
    l(90)
    f(200)
    b(400)
    f(200)
    r(90)
b(200)

#draws lines left
sh(90)
for i in range(int(graphzoom)):
    f(graphzoomdistance)
    l(90)
    f(200)
    b(400)
    f(200)
    r(90)
b(200)

#draws lines down
sh(180)
for i in range(int(graphzoom)):
    f(graphzoomdistance)
    l(90)
    f(200)
    b(400)
    f(200)
    r(90)
b(200)

#draws lines right
sh(270)
for i in range(int(graphzoom)):
    f(graphzoomdistance)
    l(90)
    f(200)
    b(400)
    f(200)

b(200)

pu()

while drawing_line == "y":
    
    print("")
    form = "placeholder"
    
    #Gets if either line or point will be drawn
    line_type = str(input("Are you drawing a line (l) or plotting a point (p) on the graph? "))
    
    #Gets Quality & Function Type
    if line_type == "l":
        quality = int(input("What do you want your line quality to be? (1-400, 1 is highest quality and vice versa, any number thats not a factor of 20 will cause glitches (any number that's not 1, 2, 4, 5, 10, or 20). Doesn't affect linear functions) "))
                
        #Please trust me this is useful later
        quality_for_i_in_range = 400/quality
                
        print("")
        type = str(input("Are you doing a linear (l), exponential (e), or quadratic (q) function? "))
                
                
        #Gets form for linear functions
        if type == "l":
            form = str(input("what form of linear function are you using? (slope-intercept (si), point-slope (ps), standard (s)) "))
                
        print("")
        
        #Gets m and b from slope-intercept form
        if form == "si":
            m = float(input("What does m (the slope) equal? "))
            b = float(input("What does b (the y-intercept) equal? "))
            
            
        #Gets m and b from point-slope form
        if form == "ps":
            y_one = float(input("what does y1 (the y value on a point of the graph) equal?"))
            m = float(input("What does m (the slope) equal? "))
            x_one = float(input("what does x1 (the x value on a point of the graph) equal?"))
            b = y_one - m * x_one
            
            
        #Gets m and b from standard form
        if form == "s":
            a = float(input("what does A equal? "))
            b_standard = float(input("what does B equal? "))
            c = float(input("what does C equal? "))
            m = -a / b_standard
            b = c / b_standard
        
        #Gets information from exponential & quadratic equations
        if type == "e":
            print("the exponential function equation is f(x) = a * b^x.")
            a_exp = float(input("What does a equal? (if there is no a then type 1) "))
            b_exp = float(input("What does b equal? "))
            vtrans_exp = float(input("How many units do you want to vertically translate your line? "))
            htrans_exp = float(input("How many units do you want to horizontally translate your line? "))
            
        #Gets information from quadratic equation
        if type == "q":
            print("the quadratic function equation is f(x) = Ax^2 + Bx + C")
            a_quad = float(input("What does A equal? (if there is no A then type 1) "))
            b_quad = float(input("What does B equal? (if there is no Bx then type 0. If there is no B then type 1.) "))
            c_quad = float(input("What does C equal? (if there is no C then type 0) "))
        
        #Sets right color, size, and position for the line
        ps(2)
        line_color = input(str("What color would you like your line to be? (hex codes allowed!) "))
        pc(line_color)
        sp(0, 0)
        
        #Tells user the line is being drawn
        print("")
        print("drawing line...")
        
        
        #Draws line if there is a linear function
        if type == "l":
            line_x = float(-200)
            mathing_x = float(-200)
            line_y = (m * mathing_x + (graphzoomdistance * b))
            sp(line_x, line_y)
            
            pd()
            
            #Draws line from left to right
            for i in range(1):
                line_x = line_x + 400
                mathing_x = mathing_x + 400
                line_y = (m * mathing_x + (graphzoomdistance * b))
                sp(line_x, line_y)
                l(90)
            
            
        #Draws line if there is an exponential function
        if type =="e":
        
            #Sets Tracy to left side of the screen
            line_x = float(-200)
            mathing_x = float(-10 - (htrans_exp * 4 / (graphzoom / 2.5)))
            line_y = ((a_exp * (b_exp ** mathing_x)) + vtrans_exp) * graphzoomdistance
            sp(line_x, line_y)
        
        
            pd()
            
            
            #Draws line from left to right
            for i in range(int(quality_for_i_in_range)):
                line_x = line_x + quality
                mathing_x = mathing_x + quality * 0.05
                line_y = ((a_exp * (b_exp ** mathing_x)) + vtrans_exp) * graphzoomdistance
                sp(line_x, line_y)
                progress = float((line_x + 200) * 0.25)
                print(str(progress) + "%")
                l(90)
        
        
        #Draws line if there is a quadratic equation
        if type =="q":
           
            #Sets Tracy to left side of the screen
            line_x = float(-200)
            mathing_x = float(-1 * graphzoom)
            line_y = ((a_quad * (mathing_x ** 2)) + (b_quad * mathing_x) + (c_quad)) * graphzoomdistance
            sp(line_x, line_y)
        
        
            pd()
            
            
            #Draws line from left to right
            for i in range(quality_for_i_in_range):
                line_x = line_x + quality
                mathing_x = mathing_x + quality * 0.05
                line_y = ((a_quad * (mathing_x ** 2)) + (b_quad * mathing_x) + (c_quad)) * graphzoomdistance
                sp(line_x, line_y)
                progress = float((line_x + 200) * 0.25)
                print(str(progress) + "%")
                l(90)
        
            #Sets position back to the center and turns color to black. Tell User the line is drawn.
            pu()
            sp(500, 0)
            sh(0)
            pc("black")
            print("")
            print("Line Drawn!")
            print("")
   
   
    #Plots Point
    if line_type == "p":
        
        #Gets x & y position of the point
        print("")
        x_point = float(input("What is the x position of your point? "))
        y_point = float(input("What is the y position of your point? "))
        pointcolor = str(input("What color do you want your point to be? (hex codes allowed!) "))
        #Moves Tracy to point
        print("")
        print("The point is being plotted...")
        pu()
        sp((x_point * 20), (y_point * 20))
        
        #Draws Point
        pc(pointcolor)
        dd(2.5)
        pd()
        sh(90)
        begin_fill()
        c(2.5)
        end_fill()
        pu()
        du(8.5)
        write(str(x_point) + ", " + str(y_point), font=("Lexend", 8), align="center")
        
        #Moves Tracy off-screen
        pu()
        sp(500, 0)
        sh(0)
        pc("black")
        
        #Tells user the point has been plotted
        print("")
        print("Point Plotted!")
        print("")


    #Checks if another line needs to be drawn
    drawing_line = input(str("Would you like to draw another line/plot another point? (y or n) "))


#Prints Closing Phrase

    print("")
